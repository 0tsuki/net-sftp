def initialize(response, text=nil)
  @response, @text = response, text
  @code = response.code
  @description = response.message
  @description = Response::MAP[@code] if @description.nil? || @description.empty?
end

def message
  m = super.dup
  m << " #{text}" if text
  m << " (#{code}, #{description.inspect})"
end
